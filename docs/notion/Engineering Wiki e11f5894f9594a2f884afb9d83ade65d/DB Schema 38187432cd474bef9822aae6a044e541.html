DB Schema

Collection: Users

Documents: (User ID) (document autoID)

Field: email, Type: string, Value: (user's email)

Field: name, Type: string, Value: (nickname randomly generated when the user signed up for an account)

Field: teamCode, Type: string, Value: (teamCode of the leaderboard the member belongs to)

Field: teamName, Type: string, Value: (teamName of the leaderboard the member belongs to)

Field: isTeamLeader, Type: boolean, Value: (whether the user is a team leader)

Field: cumulativePoints, Type: number, Value: (Summation of points from all dates)

Field: keystrokes, Type: number, Value: (number of keystrokes)

Field: timeInterval, Type: number, Value: (time the user has been spending in vscode)

Field: linesChanged, Type: number, Value: (number of lines changed)

Collection: Dates

Documents: (Date)

Field: timeInterval, Type: number, Value: (number of time in seconds)

Field: keystrokes, Type: number, Value: (number of keystrokes)

Field: linesChanged, Type: number, Value: (number of lines changed)

Field: points, Type: number, Value: (total number of points)

Collection: Leaderboards

Documents: (document autoID)

Field: teamCode, Type: string, Value: (teamCode of this leaderboard)

Field: teamLeadUserId, Type: string, Value (team lead's User ID)

Collection: Members

Documents: (User ID of member)

Field: email, Type: string, Value (team member's email)

Field: nickname, Type: string, Value (team member's nickname)
